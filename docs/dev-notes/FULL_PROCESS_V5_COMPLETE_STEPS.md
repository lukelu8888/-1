## 📋 完整业务流程概览

### 业务流程从头到尾的完整闭环：
```
客户询价 → 成本询报 → 供应商报价 → 比价择优 → 客户报价 → 审批 → 
客户接受 → 销售合同 → 定金收到 → 采购合同 → 供应商定金 → 
生产质检 → 验货完货 → 尾款催收 → 商检订舱 → 拖车装柜 → 
装柜报关 → 提单确认 → 海运费支付 → 报关开船 → 提单寄送 → 
海运跟踪 → 开票归档 → 到港清关 → 清关单证 → 清关反馈 → 
销售反馈（完整闭环）
```

app.get('/make-server-880fd43b/sync-shipment-tracking', async (c) => {
  const shipments = await getInTransitShipments();

    const tracking = await trackCargoSmartShipment(shipment.bl_no);

#### 每天检查提醒（到港前7天、免堆期前3天）
```typescript
app.get('/make-server-880fd43b/check-reminders', async (c) => {
  const shipments = await getInTransitShipments();
  const today = new Date();
  
  for (const shipment of shipments) {
    const daysUntilArrival = calculateDays(shipment.eta, today);
    
    // 到港前7天提醒
    if (daysUntilArrival === 7 && !shipment.remind_7days_sent) {
      await send7DayReminder(shipment);
      await markRemindSent(shipment.id, '7days');
    }
    
    // 免堆期前3天紧急提醒
    const daysUntilDeadline = calculateDays(shipment.free_time_deadline, today);
    if (daysUntilDeadline === 3 && !shipment.remind_3days_sent) {
      await sendUrgentReminder(shipment);
      await markRemindSent(shipment.id, '3days');
    }
  }
  
  return c.json({ success: true });
});
```

// /components/customer/ShipmentTracking.tsx
export function ShipmentTracking({ contractNo }: { contractNo: string }) {
  const [tracking, setTracking] = useState<any>(null);
  
  useEffect(() => {
    // 每30秒刷新一次
    const fetchTracking = async () => {
      const data = await fetch(`/api/shipment-tracking/${contractNo}`);
      setTracking(await data.json());
    };
    
    fetchTracking();
    const interval = setInterval(fetchTracking, 30000);
    return () => clearInterval(interval);
  }, [contractNo]);
  
  const daysUntilArrival = calculateDays(tracking.eta);
  const daysUntilFreeTimeDeadline = calculateDays(tracking.free_time_deadline);
  
  return (
    <Card>
      {/* 🔥 倒计日期大字显示 */}
      <div className="bg-gradient-to-r from-blue-500 to-blue-600 text-white p-6">
        <p className="text-6xl font-bold">{daysUntilArrival}</p>
        <p className="text-xl">天</p>
        <p>距离到港还有</p>
      </div>
      
      {/* 物流状态时间轴 */}
      <div className="space-y-4">
        {tracking.events?.map((event) => (
          <div key={event.id}>
            <p>{event.description}</p>
            <p>{event.timestamp}</p>
          </div>
        ))}
      </div>
    </Card>
  );
}
