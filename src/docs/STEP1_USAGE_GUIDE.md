## 🛠️ 技术实现

### **核心函数：routeToSalesRep**

```typescript
/**
 * 智能路由：查询或分配业务员
 */
function routeToSalesRep(
  customerName: string, 
  customerRegion: Region
): Personnel | null {
  
  // 步骤1：查询已有关系
  const mapping = getMappingByCustomer(customerName);
  if (mapping) {
    // 老客户，返回对接业务员
    return findPersonnelByName(mapping.salesRepName);
  }
  
  // 步骤2：新客户，智能分配
  const regionalSalesReps = personnelList
    .filter(p => p.role === '业务员' && p.region === customerRegion)
    .sort((a, b) => (a.workload || 0) - (b.workload || 0));
  
  const assignedSalesRep = regionalSalesReps[0];
  
  // 步骤3：保存新映射关系
  saveMapping({
    customerId: customerName,
    salesRepName: assignedSalesRep.displayName,
    assignedDate: new Date().toISOString(),
    assignedBy: 'auto'
  });
  
  return assignedSalesRep;
}
```
